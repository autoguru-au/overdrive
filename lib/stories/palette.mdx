import { ColorItem, ColorPalette, Meta } from '@storybook/addon-docs/blocks';

import { colourMap, colourMapWithoutWhite } from '../themes/base/colours';

# Palette

The AutoGuru Design System 2026 colour ramps.

export const ramp = (hue) =>
  Object.entries(colourMapWithoutWhite[hue])
    .reverse()
    // the trailing space in the key stops JS from re-sorting integer-like
    // keys ascending, keeping the 900 → 100 display order
    .reduce(
      (shades, [step, hex]) => ({ ...shades, [`${step} `]: hex.toUpperCase() }),
      {},
    );

<ColorPalette>
  <ColorItem title="Green" colors={ramp('green')} />
  <ColorItem title="Gray" colors={ramp('gray')} />
  <ColorItem title="Yellow" colors={ramp('yellow')} />
  <ColorItem title="Blue" colors={ramp('blue')} />
  <ColorItem title="Red" colors={ramp('red')} />
  <ColorItem title="White" colors={{ white: colourMap.white.toUpperCase() }} />
</ColorPalette>
